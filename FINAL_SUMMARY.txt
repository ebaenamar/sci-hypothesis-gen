========================================================================
COMMANDS AVAILABLE
========================================================================

Installation:
npm install
npm run build

Generate Single Hypothesis:
npm run generate single \
  --dataset ./data/papers.csv \
  --keywords "keyword1" "keyword2" \
  --output ./results

Generate Multiple Hypotheses:
npm run generate multiple \
  --dataset ./data/papers.csv \
  --keywords "keyword1" "keyword2" \
  --count 5

Explore Concepts:
npm run explore \
  --dataset ./data/papers.csv \
  --keywords "keyword1" "keyword2"

Find Bridge Concepts:
npm run bridges \
  --dataset ./data/papers.csv \
  --count 20

Testing:
npm test
npm run typecheck
npm run lint
